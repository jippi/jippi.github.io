 17 class ApiQueryLogListener extends CrudListener {

 19 /**
 20  * Returns a list of all events that will fire in the controller during it's lifecycle.
 21  * You can override this function to add you own listener callbacks
 22  *
 23  * We attach at priority 10 so normal bound events can run before us
 24  *
 25  * @return array
 26  */
 27     public function implementedEvents() {
 28         return array(
 29             'Crud.beforeRender' => array('callable' => 'beforeRender', 'priority' => 75)
 30         );
 31     }
 32 
 33 /**
 34  * Appends the query log to the JSON or XML output
 35  *
 36  * @param CakeEvent $event
 37  * @return void
 38  */
 39     public function beforeRender(CakeEvent $event) {
 40         if (Configure::read('debug') < 2) {
 41             return;
 42         }
 43 
 44         if (!$this->_request()->is('api')) {
 45             return;
 46         }
 47 
 48         $this->_action()->config('serialize.queryLog', 'queryLog');
 49 
 50         $queryLog = $this->_getQueryLogs();
 51         $this->_controller()->set('queryLog', $queryLog);
 52     }
 53 
 54 /**
 55  * Get the query logs for all sources
 56  *
 57  * @return array
 58  */
 59     protected function _getQueryLogs() {
 60         if (!class_exists('ConnectionManager', false)) {
 61             return array();
 62         }
 63 
 64         $sources = $this->_getSources();
 65         $queryLog = array();
 66         foreach ($sources as $source) {
 67             $db = $this->_getSource($source);
 68 
 69             if (!method_exists($db, 'getLog')) {
 70                 continue;
 71             }
 72 
 73             $queryLog[$source] = $db->getLog(false, false);
 74         }
 75 
 76         return $queryLog;
 77     }
 78 
 79 /**
 80  * Get a list of sources defined in database.php
 81  *
 82  * @codeCoverageIgnore
 83  * @return array
 84  */
 85     protected function _getSources() {
 86         return ConnectionManager::sourceList();
 87     }
 88 
 89 /**
 90  * Get a specific data source
 91  *
 92  * @codeCoverageIgnore
 93  * @param string $source Datasource name
 94  * @return DataSource
 95  */
 96     protected function _getSource($source) {
 97         return ConnectionManager::getDataSource($source);
 98     }
